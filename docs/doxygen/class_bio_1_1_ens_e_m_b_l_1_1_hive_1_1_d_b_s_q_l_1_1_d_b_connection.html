sub url {
my ($self, $psw_env_var_name) = @_;
my $url = $self->driver . '://';
if($self->username) {
$url .= $self->username;
if(my $psw_expression = $self->password) {
if($psw_env_var_name) {
$ENV{$psw_env_var_name} = $psw_expression;
$psw_expression = '${'.$psw_env_var_name.'}';
}
$url .= ':'.$psw_expression if($psw_expression);
}
$url .= '@';
}
if($self->host) {
$url .= $self->host;
if($self->port) {
$url .= ':'.$self->port;
}
}
$url .= '/' . $self->dbname;
return $url;
}
