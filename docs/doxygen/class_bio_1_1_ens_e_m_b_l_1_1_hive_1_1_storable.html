Description

    Storable is a base class for anything that can be stored.
    It provides two getters/setters: dbID() and adaptor().

sub new {
my $class = shift @_;
my $self = bless {}, $class;
while(my ($method,$value) = splice(@_,0,2)) {
if(defined($value)) {
$self->$method($value);
}
}
return $self;
